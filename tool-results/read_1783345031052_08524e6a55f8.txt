    81тЖТ    81тЖТ    return new Promise((resolve, reject) => {
    82тЖТ    82тЖТ      const reader = new FileReader()

    86тЖТ    86тЖТ          const canvas = document.createElement('canvas')
    87тЖТ    87тЖТ          let { width, height } = img
    88тЖТ    88тЖТ          if (width > MAX_DIMENSION || height > MAX_DIMENSION) {
    89тЖТ    89тЖТ            if (width > height) {
    90тЖТ    90тЖТ              height = Math.round((height * MAX_DIMENSION) / width)
    91тЖТ    91тЖТ              width = MAX_DIMENSION
    92тЖТ    92тЖТ            } else {

   333тЖТ   333тЖТfunction generateSlug(title: string): string {
   334тЖТ   334тЖТ  const roMap: Record<string, string> = {
   335тЖТ   335тЖТ    '─Г': 'a', '├в': 'a', '├о': 'i', '╚Щ': 's', '╚Ы': 't',

   337тЖТ   337тЖТ  }
   338тЖТ   338тЖТ  return title
