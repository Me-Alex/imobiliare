    74тЖТ    74тЖТ  const [urlInput, setUrlInput] = useState('')
    75тЖТ    75тЖТ  const fileInputRef = useRef<HTMLInputElement>(null)
    76тЖТ    76тЖТ  const MAX_FILE_SIZE = 10 * 1024 * 1024 // 10MB raw input
    77тЖТ    77тЖТ  const MAX_IMAGES = 15
    78тЖТ    78тЖТ  const MAX_DIMENSION = 1280 // max width/height in px
    79тЖТ    79тЖТ  const JPEG_QUALITY = 0.75
    80тЖТ    80тЖТ
    81тЖТ    81тЖТ  const compressImage = useCallback((file: File): Promise<string> => {
    82тЖТ    82тЖТ    return new Promise((resolve, reject) => {
    83тЖТ    83тЖТ      const reader = new FileReader()
    84тЖТ    84тЖТ      reader.onload = (e) => {
    85тЖТ    85тЖТ        const img = new Image()
    86тЖТ    86тЖТ        img.onload = () => {
    87тЖТ    87тЖТ          const canvas = document.createElement('canvas')
    88тЖТ    88тЖТ          let { width, height } = img
    89тЖТ    89тЖТ          if (width > MAX_DIMENSION || height > MAX_DIMENSION) {
    90тЖТ    90тЖТ            if (width > height) {
    91тЖТ    91тЖТ              height = Math.round((height * MAX_DIMENSION) / width)
    92тЖТ    92тЖТ              width = MAX_DIMENSION
    93тЖТ    93тЖТ            } else {
    94тЖТ    94тЖТ              width = Math.round((width * MAX_DIMENSION) / height)
    95тЖТ    95тЖТ              height = MAX_DIMENSION
    96тЖТ    96тЖТ            }
    97тЖТ    97тЖТ          }
    98тЖТ    98тЖТ          canvas.width = width
    99тЖТ    99тЖТ          canvas.height = height
   100тЖТ   100тЖТ          const ctx = canvas.getContext('2d')
   101тЖТ   101тЖТ          if (!ctx) { reject(new Error('Canvas not supported')); return }
   102тЖТ   102тЖТ          ctx.drawImage(img, 0, 0, width, height)
   103тЖТ   103тЖТ          resolve(canvas.toDataURL('image/jpeg', JPEG_QUALITY))
   104тЖТ   104тЖТ        }
   105тЖТ   105тЖТ        img.onerror = () => reject(new Error('Failed to load image'))
   106тЖТ   106тЖТ        img.src = e.target?.result as string
   107тЖТ   107тЖТ      }
   108тЖТ   108тЖТ      reader.onerror = () => reject(new Error('Failed to read file'))
   109тЖТ   109тЖТ      reader.readAsDataURL(file)
   110тЖТ   110тЖТ    })

   122тЖТ   122тЖТ        return
   123тЖТ   123тЖТ      }

   139тЖТ   139тЖТ        onChange([...urls, ...validResults])

   142тЖТ   142тЖТ      setIsReading(false)
   143тЖТ   143тЖТ    }

   163тЖТ   163тЖТ    if (files.length > 0) {
   164тЖТ   164тЖТ      readFilesAsDataUrls(files)
   165тЖТ   165тЖТ    }
   166тЖТ   166тЖТ  }, [readFilesAsDataUrls])
   167тЖТ   167тЖТ
   168тЖТ   168тЖТ  const handleFileSelect = useCallback((e: React.ChangeEvent<HTMLInputElement>) => {

   231тЖТ   231тЖТ              Alege Fotografii

   334тЖТ   334тЖТfunction generateSlug(title: string): string {
   335тЖТ   335тЖТ  const roMap: Record<string, string> = {
